\section{Week 8}

\begin{exercise} 
  Use Zorn's lemma to show the existence of a transcendence base $S$ of any 
  extension $L/K$. ($S$ may equal to $\emptyset$)
\end{exercise}

\begin{exercise}
  Given $L/M, M/K$, show that $\mathrm{tr} \deg_K L = \mathrm{tr} \deg_M L
  + \mathrm{tr} \deg_K M$.
\end{exercise}
