\section{Week 11}

\begin{exercise} \mbox{}
  \begin{enumerate}[(a)]
    \item Show that $\Set{ y-x^2, z-x^3 }$ is not a Gr\"obner basis for $x>y>z$.
    \item Find a Gr\"ober basis for $\Gen{ y-x^2, z-x^3 }$.
  \end{enumerate}
\end{exercise}

\begin{exercise} \mbox{}
  \begin{enumerate}[(a)]
    \item Find a Gr\"obner basis of $I = \Gen{ -x^3+y, x^2y - y^2 }$.
    \item Check $x^6 - x^5y \overset{?}{\in} I$.
  \end{enumerate}
\end{exercise}
